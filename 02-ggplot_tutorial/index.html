    Create a scatterplot

    ggplot(data = iris_clean, aes(x = sepal_length, y = sepal_width, color = species)) +
  geom_point(size = 3) +
  scale_color_viridis_d() +
  labs(x = "Sepal length",
       y = "Sepal width",
       color = "Species") +
  theme_light(base_size = 12)
